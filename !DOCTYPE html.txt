<script>
  const chatBox = document.getElementById("chatBox");
  const userInput = document.getElementById("userInput");

  userInput.addEventListener("keypress", function(e) {
    if (e.key === "Enter" && userInput.value.trim() !== "") {
      addMessage(userInput.value, "user");
      generateResponse(userInput.value.toLowerCase());
      userInput.value = "";
    }
  });

  function addMessage(text, sender) {
    const msg = document.createElement("div");
    msg.className = sender;
    msg.innerText = text;
    chatBox.appendChild(msg);
    chatBox.scrollTop = chatBox.scrollHeight;
  }

  function generateResponse(input) {
    let response = "";

    if (input.includes("stress") || input.includes("tired")) {
      response =
        "It sounds like you’re under pressure. Try this: inhale slowly for 4 seconds, hold for 4, exhale for 6. Small pauses help reset the mind.";
    }
    else if (input.includes("sad") || input.includes("down")) {
      response =
        "Feeling low can be heavy. Writing one thing that bothered you today can sometimes make it feel more manageable.";
    }
    else if (input.includes("angry") || input.includes("mad")) {
      response =
        "Anger usually means something important feels threatened. Take a short pause and step away from the trigger if possible.";
    }
    else if (input.includes("happy") || input.includes("good")) {
      response =
        "That’s good to hear. Noticing positive moments helps strengthen emotional balance.";
    }
    else if (input.includes("hopeless") || input.includes("worthless")) {
      response =
        "You deserve support beyond a chatbot. Please consider talking to a trusted adult, teacher, or mental health professional.";
    }
    else {
      response =
        "Thank you for sharing. Can you describe your feeling a little more?";
    }

    setTimeout(() => addMessage(response, "bot"), 500);
  }
</script>
